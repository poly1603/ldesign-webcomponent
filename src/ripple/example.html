📚 使用示例

<!-- 基础用法 --> <button class="btn"> <l-ripple></l-ripple> 点击我 </button> <!-- 自定义配置 --> <button class="btn"> <l-ripple ripple-color="#ff6b6b" ripple-opacity="0.3" ripple-duration="800" ripple-centered="true"> </l-ripple> 自定义波纹 </button> <!-- JavaScript 控制 --> <script> const ripple = document.querySelector('l-ripple'); // 手动触发波纹 ripple.trigger(x, y); // 更新配置 ripple.updateOptions({ color: '#667eea', duration: 600 }); // 启用/禁用 ripple.enable(); ripple.disable(); </script>
