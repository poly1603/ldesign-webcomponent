代码示例

<ldesign-draggable src="image.jpg" enable-grid grid-size="25" show-indicators enable-history show-minimap smooth-bounds show-controls show-performance ></ldesign-draggable> <script> // 获取组件实例 const draggable = document.querySelector('ldesign-draggable'); // 设置预设视角 draggable.presetViews = [ { name: '全景', scale: 1, rotate: 0, offsetX: 0, offsetY: 0 }, { name: '细节', scale: 3, rotate: 0, offsetX: 50, offsetY: -50 } ]; // 控制方法 draggable.reset(); // 重置 draggable.zoomTo(2); // 缩放到2倍 draggable.rotateBy(90); // 旋转90度 draggable.undo(); // 撤销 draggable.redo(); // 重做 </script>
